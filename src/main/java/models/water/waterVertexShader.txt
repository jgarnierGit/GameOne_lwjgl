#version 400 core

in vec3 position;

out vec4 clipSpace;

uniform mat4 transformationMatrix;
uniform mat4 viewMatrix;
uniform mat4 projectionMatrix;


void main(void) {

	clipSpace = projectionMatrix * viewMatrix * transformationMatrix * vec4(position.x, position.y -1.0, position.z, 1.0);
	gl_Position = clipSpace;
 
}
